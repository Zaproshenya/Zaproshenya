## Bug 1: Invitations don't load — `ReferenceError` on `statuses` (CRITICAL)

**File:** `js/pages/home.js:19-43`

**Problem:** Variable `statuses` is declared with `const` inside the first `if (statusSnap.exists())` block (line 21) but referenced in the second `if (statusSnap.exists())` block (line 40). Since `const` is block-scoped, this causes `ReferenceError: statuses is not defined`. The error is caught by `app.js:209` but `loaded` and `_loaded` remain `false`, so the page shows a perpetual spinner.

**Fix:** Move `const statuses = statusSnap.val();` before the first `if` block (after line 20), so it's accessible in both blocks.

```js
// BEFORE (broken):
if (statusSnap.exists()) {
  const statuses = statusSnap.val();
  invites.forEach(inv => { ... });
}
// ... 
if (statusSnap.exists()) {
  incomingInvites = incomingInvites.filter(n => {
    const s = statuses[n.inviteId]; // ReferenceError!
  });
}

// AFTER (fixed):
const statuses = statusSnap.exists() ? statusSnap.val() : {};
if (statusSnap.exists()) {
  invites.forEach(inv => { ... });
}
// ...
if (statusSnap.exists()) {
  incomingInvites = incomingInvites.filter(n => {
    const s = statuses[n.inviteId]; // OK
  });
}
```

## Bug 2: Friend invite filter missing `inviteId` check

**File:** `js/pages/friends.js:33-36`

**Problem:** The `friendInvites` filter doesn't check `n.inviteId`, unlike `home.js:33-35`. Malformed notifications without `inviteId` get included, inflating the count and causing broken navigation.

**Fix:** Add `n.inviteId` to the filter:

```js
friendInvites = notifs.filter(n =>
  (n.type === 'invite' || n.type === 'group-invite') && n.inviteId && !n.read &&
  now - n.createdAt < SEVEN_DAYS
);
```

## Bug 4: Auto-fill friend name when selecting in create.js

**File:** `js/pages/create.js:276-293`

**Problem:** `toggleFriend()` in personal mode sets `selectedFriends` but doesn't populate the `f-to` input with the friend's name or make it readonly. Submit button stays disabled because `chk()` requires `f-to` to have a value.

**Fix:** In `toggleFriend()`, when selecting a friend in personal mode, look up the friend object and populate the input. When deselecting, clear it.

```js
function toggleFriend(uid, ctx, el) {
    saveFormState();
    if (ctx === 'personal') {
      const wasSelected = selectedFriends.includes(uid);
      selectedFriends = wasSelected ? [] : [uid];
      document.querySelectorAll('#cform .pill.on').forEach(b => b.classList.remove('on'));
      
      const toInput = document.getElementById('f-to');
      if (!wasSelected) {
        el.classList.add('on');
        // Auto-fill friend name
        const friend = friends.find(f => f.uid === uid);
        if (friend && toInput) {
          toInput.value = friend.name;
          toInput.readOnly = true;
          toInput.style.opacity = '0.7';
          formState.to = friend.name;
        }
      } else {
        // Clear friend name
        if (toInput) {
          toInput.value = '';
          toInput.readOnly = false;
          toInput.style.opacity = '';
          formState.to = '';
        }
      }
    } else {
      // ... group mode unchanged
    }
    chk();
  }
```

## Bug 5: Friend menu clips on mobile — `overflow: hidden` on `.friend-list`

**File:** `css/pages.css:163-168`

**Problem:** `.friend-list` has `overflow: hidden` which clips the absolutely-positioned `.friend-menu` when it extends past the container bottom.

**Fix:** Use `clip-path` instead of `overflow: hidden` to preserve border-radius while allowing the menu to escape. Or alternatively, switch `.friend-menu` to `position: fixed` with computed coordinates.

**Approach: Switch to `position: fixed` for the menu.**

In `js/pages/friends.js`, update `toggleMenu` to compute fixed position from the button's bounding rect:

```js
function toggleMenu(e, uid) {
    e.stopPropagation();
    if (openMenuUid === uid) {
      closeMenu();
      return;
    }
    closeMenu();
    const row = e.target.closest('.friend-row');
    if (!row) return;
    const f = friends.find(x => x.uid === uid);
    if (!f) return;
    
    const rect = e.target.getBoundingClientRect();
    const menu = document.createElement('div');
    menu.className = 'friend-menu';
    // Position fixed relative to viewport
    menu.style.position = 'fixed';
    menu.style.top = (rect.bottom + 4) + 'px';
    menu.style.right = (window.innerWidth - rect.right) + 'px';
    menu.style.zIndex = '999';
    // Check if menu would overflow bottom
    const menuHeight = 120; // approximate
    if (rect.bottom + menuHeight > window.innerHeight) {
      menu.style.top = 'auto';
      menu.style.bottom = (window.innerHeight - rect.top + 4) + 'px';
    }
    
    menu.innerHTML = `...`;
    document.body.appendChild(menu);
    openMenuUid = uid;
    setTimeout(() => document.addEventListener('click', closeMenu, { once: true }), 10);
  }
```

In `css/pages.css`, update `.friend-menu` styles:

```css
.friend-menu {
  /* Remove position: absolute; use fixed instead */
  min-width: 160px;
  box-shadow: 0 8px 30px rgba(0,0,0,.15);
}
```

## Bug 6: Menu doesn't close on re-click of three dots

**File:** `js/pages/friends.js:120-143`

**Problem:** `toggleMenu` always calls `closeMenu()` then immediately recreates the menu. There's no guard checking if the same menu is already open.

**Fix:** Add early-return guard (combined with Bug 5 fix above):

```js
function toggleMenu(e, uid) {
    e.stopPropagation();
    if (openMenuUid === uid) {
      closeMenu();
      return;
    }
    closeMenu();
    // ... rest of menu creation
```
